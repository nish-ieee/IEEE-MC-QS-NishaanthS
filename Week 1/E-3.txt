//1. Two Sum

class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int, int> visited;

        for(int i = 0; i < nums.size(); i++) {
            int num = target - nums[i];
            if(visited.count(num)) return {visited[num], i};
            visited[nums[i]] = i;
        }

        return {};
    }
};
