//70. Climbing Stairs

class Solution {
public:
    int climbStairs(int n) {
        unordered_map<int, int> counts{{1, 1}, {2, 2}};
        int i = 3;

        while(i <= n) {
            counts[i] = counts[i - 1] + counts[i - 2];
            i++;
        }
        
        return counts[n];
    }
};
